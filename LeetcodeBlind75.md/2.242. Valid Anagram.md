```python
class Solution:
    def isAnagram(self, s: str, t: str) -> bool:
        if len(s)!=len(t):
            return False
        hashs,hasht={},{}
        for i in range(len(s)):
            hashs[s[i]]=1+hashs.get(s[i],0)
            hasht[t[i]]=1+hasht.get(t[i],0)
        return hashs==hasht
  ```
